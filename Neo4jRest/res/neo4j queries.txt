create domain

create link
CREATE (n:Link { name:'Link1', link:'www.wikipedia1.com' })

create relation between a link and a domain
match (d:Domain), (l:Link) where d.name='Wikipedia' and l.name='Link1' create (l)-[r:PART_OF]->(d)

create relation between two links
match (l1:Link), (l2:Link) where l1.name='Link1' and l2.name='Link3' create (l1)-[r:LINKS_TO]->(l2)

read a domain's links
match (l:Link)-[r]-> (d:Domain{name:'Wikipedia'}) return r

read all links and relationships
match (l:Link)-[r]-> () return r

delete all nodes and relationships
MATCH (n) OPTIONAL MATCH (n)-[r]-() DELETE n,r
